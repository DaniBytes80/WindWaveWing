Políticas de seguridad de RSL de Supabase a nivel de perfiles.
-- 1. Habilitar RLS en la tabla (si no lo está ya)
ALTER TABLE public."Perfiles" ENABLE ROW LEVEL SECURITY;

-- 2. Regla para CONSULTAR (SELECT)
-- Permite que cualquier usuario registrado (o incluso visitantes) vea los perfiles
-- Útil para buscadores de spots/amigos
CREATE POLICY "Los perfiles son visibles para usuarios autenticados" 
ON public."Perfiles" 
FOR SELECT 
TO authenticated 
USING (true);

-- 3. Regla para INSERTAR (INSERT)
-- El usuario solo puede crear un perfil si el ID coincide con su ID de autenticación
CREATE POLICY "Los usuarios solo crean su propio perfil" 
ON public."Perfiles" 
FOR INSERT 
TO authenticated 
WITH CHECK (auth.uid() = id);

-- 4. Regla para MODIFICAR (UPDATE)
-- Solo el dueño del perfil (donde auth.uid() coincide con id) puede editar
CREATE POLICY "Los usuarios solo editan su propio perfil" 
ON public."Perfiles" 
FOR UPDATE 
TO authenticated 
USING (auth.uid() = id) 
WITH CHECK (auth.uid() = id);

-- 5. Regla para ELIMINAR (DELETE)
-- Solo el dueño del perfil puede borrar su cuenta
CREATE POLICY "Los usuarios solo borran su propio perfil" 
ON public."Perfiles" 
FOR DELETE 
TO authenticated 
USING (auth.uid() = id);


--  Trigger en la tabla de auth.users que crea un registro en la tabla public.Perfiles
-- con el id,nombre y (email o telefono) para poder registrarse. Además de rol y avatar si los hay.
-- En el caso de activar el registro mediante google o Apple tambien se permite.

create or replace function public.handle_new_user()
returns trigger as $$
begin
  insert into public."Perfiles" (id, email, nombre, rol, telefono, avatar_url)
  values (
    new.id, 
    new.email, 
     coalesce(new.raw_user_meta_data->>'full_name', new.raw_user_meta_data->>'name', 'Usuario Nuevo'),
    'usuario',
    new.phone,
    new.raw_user_meta_data->>'avatar_url'
  );
  return new;
end;
$$ language plpgsql security definer;

-- Creación de políticas de archivos 
create policy "Cualquiera puede ver los avatares"
on storage.objects
for select
using (bucket_id = 'avatars');

create policy "El usuario solo puede subir su propio avatar"
on storage.objects
for insert
to authenticated
with check (
  bucket_id = 'avatars'
  and (storage.foldername(name))[1] = auth.uid()::text
);

create policy "El usuario solo puede actualizar su propio avatar"
on storage.objects
for update
to authenticated
using (
  bucket_id = 'avatars'
  and (storage.foldername(name))[1] = auth.uid()::text
)
with check (
  bucket_id = 'avatars'
  and (storage.foldername(name))[1] = auth.uid()::text
);

create policy "El usuario solo puede borrar su propio avatar"
on storage.objects
for delete
to authenticated
using (
  bucket_id = 'avatars'
  and (storage.foldername(name))[1] = auth.uid()::text
);

create table public."AlertasUsuario" (
    id uuid not null default gen_random_uuid(),
    user_id uuid not null,
    spot_id uuid not null,
    
    -- Condición meteorológica a evaluar
    parametro text not null,          -- viento, racha, ola, direccion, periodo, etc.
    operador text not null,           -- >, <, >=, <=, =
    valor numeric not null,           -- valor numérico a comparar
    
    -- Opcional: material preferido para esta alerta
    material_id uuid null,
    
    -- Estado de la alerta
    activa boolean not null default true,
    
    -- Texto opcional para mostrar al usuario
    nombre text null,
    descripcion text null,
    
    fecha_creacion timestamp without time zone not null default now(),
    
    constraint AlertasUsuario_pkey primary key (id),
    
    constraint AlertasUsuario_user_id_fkey
        foreign key (user_id) references "Perfiles" (id) on delete cascade,
    
    constraint AlertasUsuario_spot_id_fkey
        foreign key (spot_id) references spot (id) on delete cascade,
    
    constraint AlertasUsuario_material_id_fkey
        foreign key (material_id) references "MaterialUsuario" (id) on delete set null
);

-- ============================================================
--  FUNCIÓN: obtener_grid_actual
--  Devuelve el snapshot más reciente del grid.
--  La app llama a esta función RPC en vez de hacer un SELECT
--  con filtros complejos → más eficiente y seguro.
-- ============================================================
CREATE OR REPLACE FUNCTION public.obtener_grid_actual(
    p_min_lat NUMERIC DEFAULT -90,
    p_max_lat NUMERIC DEFAULT  90,
    p_min_lng NUMERIC DEFAULT -180,
    p_max_lng NUMERIC DEFAULT  180
)
RETURNS TABLE (
    lat              NUMERIC,
    lng              NUMERIC,
    velocidad_viento NUMERIC,
    direccion_viento NUMERIC,
    racha_viento     NUMERIC,
    temperatura      NUMERIC,
    precipitacion    NUMERIC,
    altura_ola       NUMERIC,
    direccion_ola    NUMERIC,
    periodo_ola      NUMERIC
)
LANGUAGE sql
STABLE
AS $$
    SELECT
        lat, lng,
        velocidad_viento, direccion_viento, racha_viento,
        temperatura, precipitacion,
        altura_ola, direccion_ola, periodo_ola
    FROM public.clima_grid
    WHERE
        -- Solo el snapshot más reciente
        fecha_hora = (SELECT MAX(fecha_hora) FROM public.clima_grid)
        -- Solo el bbox visible del mapa
        AND lat BETWEEN p_min_lat AND p_max_lat
        AND lng BETWEEN p_min_lng AND p_max_lng
    ORDER BY lat, lng;
$$;

-- Permisos para que la app pueda llamar a la función
GRANT EXECUTE ON FUNCTION public.obtener_grid_actual TO anon;
GRANT EXECUTE ON FUNCTION public.obtener_grid_actual TO authenticated;
