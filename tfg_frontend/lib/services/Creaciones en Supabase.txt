Una vez creado el proyecto en Supabase, se crean las siguientes tablas. 

Tabla de Usuario:
La tabla de usuario está vinculada con la auth.users del sistema de Supabase por motivos de gestión
de seguridad de los mismos.

create table public."Perfiles" (
  nombre character varying null,
  email character varying not null,
  rol character varying not null,
  fecha_registro timestamp with time zone not null default now(),
  telefono character varying null,
  surf boolean null default false,
  kite_surf boolean null default false,
  windsurf boolean null default false,
  wing boolean null default false,
  sail boolean null default false,
  id uuid not null default auth.uid (),
  avatar_url text null,
  constraint Perfiles_pkey primary key (id),
  constraint Perfiles_email_key unique (email)
) TABLESPACE pg_default;

  spot_id uuid null,

  velocidad_viento double precision null,
  direccion_viento text null,
  racha_viento double precision null,
  altura_ola double precision null,
  periodo_ola double precision null,
  direccion_ola text null,
  temperatura double precision null,
  humedad double precision null,
  probabilidad_lluvia double precision null,

  constraint clima_spot_id_fkey foreign KEY (spot_id) references spot (id) on delete CASCADE

ALTER TABLE public.clima ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Allow select for authenticated"
ON clima
FOR SELECT
TO authenticated
USING (true); -- O verifica auth.uid() = user_id

ON clima
FOR UPDATE
TO authenticated

-- Creamos la tabla de favoritos (Relación muchos a muchos)
-- Asegúrate de que la tabla "Perfiles" tenga una columna 'id' que sea UUID
create table public.favoritos_spots (
  id uuid primary key default uuid_generate_v4(),
  perfil_id uuid references public."Perfiles"(id) on delete cascade not null,
  spot_id uuid references public.spot(id) on delete cascade not null,
  created_at timestamp with time zone default now() not null,
  unique(perfil_id, spot_id)
);

Políticas de seguridad de RSL de Supabase a nivel de perfiles.
-- 1. Habilitar RLS en la tabla (si no lo está ya)
ALTER TABLE public."Perfiles" ENABLE ROW LEVEL SECURITY;

-- 2. Regla para CONSULTAR (SELECT)
-- Permite que cualquier usuario registrado (o incluso visitantes) vea los perfiles
-- Útil para buscadores de spots/amigos
CREATE POLICY "Los perfiles son visibles para usuarios autenticados" 
ON public."Perfiles" 
FOR SELECT 
TO authenticated 
USING (true);

-- 3. Regla para INSERTAR (INSERT)
-- El usuario solo puede crear un perfil si el ID coincide con su ID de autenticación
CREATE POLICY "Los usuarios solo crean su propio perfil" 
ON public."Perfiles" 
FOR INSERT 
TO authenticated 
WITH CHECK (auth.uid() = id);

-- 4. Regla para MODIFICAR (UPDATE)
-- Solo el dueño del perfil (donde auth.uid() coincide con id) puede editar
CREATE POLICY "Los usuarios solo editan su propio perfil" 
ON public."Perfiles" 
FOR UPDATE 
TO authenticated 
USING (auth.uid() = id) 
WITH CHECK (auth.uid() = id);

-- 5. Regla para ELIMINAR (DELETE)
-- Solo el dueño del perfil puede borrar su cuenta
CREATE POLICY "Los usuarios solo borran su propio perfil" 
ON public."Perfiles" 
FOR DELETE 
TO authenticated 
USING (auth.uid() = id);


--  Trigger en la tabla de auth.users que crea un registro en la tabla public.Perfiles
-- con el id,nombre y (email o telefono) para poder registrarse. Además de rol y avatar si los hay.
-- En el caso de activar el registro mediante google o Apple tambien se permite.

create or replace function public.handle_new_user()
returns trigger as $$
begin
  insert into public."Perfiles" (id, email, nombre, rol, telefono, avatar_url)
  values (
    new.id, 
    new.email, 
     coalesce(new.raw_user_meta_data->>'full_name', new.raw_user_meta_data->>'name', 'Usuario Nuevo'),
    'usuario',
    new.phone,
    new.raw_user_meta_data->>'avatar_url'
  );
  return new;
end;
$$ language plpgsql security definer;
